Table.add_row() Example
\begin{python}
    # some sample data
    headers = ['Name', 'Age (Yr)', 'Height (m)']
    data = [
        ['John', 20, 1.8],
        ['Mary', 21, 1.7],
        ['Peter', 22, 1.9],
    ]

    t = Table(data, headers = headers, caption = 'Sample table')
    t.add_row(['Sam', 23, 1.6])

    export(t)
\end{python}
